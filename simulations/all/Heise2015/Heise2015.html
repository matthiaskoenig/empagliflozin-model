from typing import Dict

from sbmlsim.data import DataSet, load_pkdb_dataframe
from sbmlsim.fit import FitMapping, FitData
from sbmlutils.console import console
from sbmlsim.plot import Axis, Figure
from sbmlsim.simulation import Timecourse, TimecourseSim

from pkdb_models.models.empagliflozin.experiments.base_experiment import EmpagliflozinSimulationExperiment
from pkdb_models.models.empagliflozin.experiments.metadata import Tissue, Route, Dosing, ApplicationForm, Health, \
    Fasting, EmpagliflozinMappingMetaData, Coadministration
from pkdb_models.models.empagliflozin.helpers import run_experiments


class Heise2015(EmpagliflozinSimulationExperiment):
    """Simulation experiment of Heise2015."""

    fpg = 10  # [mM] (t2dm, actual value not reported)
    bodyweight = 77.8  # [kg]
    gfr = 90  # [ml/min] (t2dm)

    dset_ids = [
        "emp25",
        "emp25,hctz25",
        "emp25,tor5",
    ]
    colors = {
        "emp25": "black",
        "emp25,hctz25": "tab:blue",
        "emp25,tor5": "tab:orange",
    }

    def datasets(self) -> Dict[str, DataSet]:
        dsets = {}
        for fig_id in ["Fig2"]:
            df = load_pkdb_dataframe(f"{self.sid}_{fig_id}", data_path=self.data_path)
            for label, df_label in df.groupby("label"):
                dset = DataSet.from_df(df_label, self.ureg)

                dsets[f"{label}"] = dset

        # console.print(dsets)
        # console.print(dsets.keys())
        return dsets

    def simulations(self) -> Dict[str, TimecourseSim]:
        Q_ = self.Q_
        tcsims = {}
        # Multiple dosing (5 doses)

        tc0 = Timecourse(
            start=0,
            end=24 * 60,  # [min]
            steps=500,
            changes={
                **self.default_changes(),
                "BW": Q_(self.bodyweight, "kg"),
                "[KI__fpg]": Q_(self.fpg, "mM"),
                # "KI__f_renal_function": Q_(self.gfr, "dimensionless")
                "PODOSE_emp": Q_(25, "mg"),
            },
        )
        tc1 = Timecourse(
            start=0,
            end=24 * 60,  # [min]
            steps=500,
            changes={
                "PODOSE_emp": Q_(25, "mg"),
            },
        )
        tc2 = Timecourse(
            start=0,
            end=25 * 60,  # [min]
            steps=500,
            changes={
                "PODOSE_emp": Q_(25, "mg"),
            },
        )

        tcsims[f"po_emp25"] = TimecourseSim(
            [tc0] + [tc1 for _ in range(3)] + [tc2],
            time_offset=-4*24*60,
        )
        # console.print(tcsims.keys())
        return tcsims

    def fit_mappings(self) -> Dict[str, FitMapping]:
        mappings = {}
        for dset in self.dset_ids:
            coadministration = Coadministration.NONE
            if "hctz25" in dset:
                coadministration = Coadministration.HCTZ
            elif "tor5" in dset:
                coadministration = Coadministration.TORASEMIDE

            mappings[f"fm_po_emp25_{dset}"] = FitMapping(
                self,
                reference=FitData(
                    self,
                    dataset=dset,
                    xid="time",
                    yid="mean",
                    yid_sd="mean_sd",
                    count="count",
                ),
                observable=FitData(
                    self, task=f"task_po_emp25", xid="time", yid=f"[Cve_emp]",
                ),
                metadata=EmpagliflozinMappingMetaData(
                    tissue=Tissue.PLASMA,
                    route=Route.PO,
                    application_form=ApplicationForm.TABLET,
                    dosing=Dosing.MULTIPLE,
                    health=Health.T2DM,
                    fasting=Fasting.NR,
                    coadministration=coadministration,
                ),
            )
        # console.print(mappings)
        return mappings

    def figures(self) -> Dict[str, Figure]:
        fig = Figure(
            experiment=self,
            sid="Fig2",
            name=f"{self.__class__.__name__} (T2DM)",
        )
        plots = fig.create_plots(xaxis=Axis(self.label_time, unit=self.unit_time), legend=True)
        plots[0].set_yaxis(self.label_emp_plasma, unit=self.unit_emp)
        plots[0].xaxis.min = -5
        plots[0].xaxis.max = 25

        # simulation
        plots[0].add_data(
            task=f"task_po_emp25",
            xid="time",
            yid=f"[Cve_emp]",
            label=f"25 mg Emp",
            color="black",
        )
        coadmin_labels = {
            "emp25": None,
            "emp25,hctz25": "HCTZ",
            "emp25,tor5": "Tor",
        }

        for dset in self.dset_ids:
            coadmin = coadmin_labels[dset]
            label = "25 mg Emp" if coadmin is None else f"25 mg Emp + {coadmin}"

            plots[0].add_data(
                dataset=dset,
                xid="time",
                yid="mean",
                yid_sd="mean_sd",
                count="count",
                label=label,
                color=self.colors[dset],
            )

        return {
            fig.sid: fig,
        }


if __name__ == "__main__":
    run_experiments(Heise2015, output_dir=Heise2015.__name__)
